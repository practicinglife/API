REST SDK Definitions

Methods

ResetCookieAuthentication - Wipes out the member password hash to reset from new username and password in the next invoke request

SetCompanyName Sets the target company name, must match the company name in login database

SetCookieAuthentication Sets the members username and members password hash values.

SetPublicPrivateKey Sets the private and public key values used for authentication.

SetSite The target URL of the ConnectWise PSA application

SetTimeout Set the max length to wait for an executed http request to respond

SetupForCookieAuthentication - Sets the username and password of the member. The client will automatically grab the password hash when the first request happens.

SetUserAgent Sets the user-agent value for incoming http requests. Defaults to ConnectWise-DotNet-SDK.

UseHttp Set this method to configure the client to use http protocol (Defaults to https)

Properties

DateFormat - Use ISO 8601 format for date and datetime. [yyyy-MM-ddTHH:mm:ssZ]

ConnectwiseAppId The integrators ConnectWise PSA Application Id

Site The ConnectWise PSA Application URL

CodeBase The preferred code base to be used when executing the request

Username The memberIdentifier of the ConnectWise PSA User being impersonated

Password The ConnectWise PSA Members application password

MemberHash The ConnectWise PSA Members password hash value

PublicKey Alternative method of authentication. *Can be deleted through UI

PrivateKey Used with Public Key for authentication.

Scheme Defaults to Https:// unless UseHttp() is invoked.

UsingCookieAuth Set to true if using cookie authentication for requests

214        /// <summary>
215        /// Invoke an API request
216        /// </summary>
217        public async Task<Response> InvokeAPIAsync(string path, string method, Dictionary<string, string> queryParams, object body)
218        {
219            using(var httpClient = await getHttpClient())
220            {
221                HttpResponseMessage response;
222                var formatter = GetJsonMediaFormatter();
223                var upperMethod = method.ToUpper();
224                var fullPath = await getFullPath(path, queryParams);
225                if (upperMethod.Equals("GET", StringComparison.CurrentCultureIgnoreCase))
226                {
227                    response = httpClient.GetAsync(fullPath).Result;
228                }
229                else if (upperMethod.Equals("POST", StringComparison.CurrentCultureIgnoreCase))
230                {
231                    response = httpClient.PostAsync(fullPath, body, formatter, DefaultContentType).Result;
232                }
233                else if (upperMethod.Equals("PUT", StringComparison.CurrentCultureIgnoreCase))
234                {
235                    response = httpClient.PutAsync(fullPath, body, formatter, DefaultContentType).Result;
236                }
237                else if (upperMethod.Equals("DELETE", StringComparison.CurrentCultureIgnoreCase))
238                {
239                    response = httpClient.DeleteAsync(fullPath).Result;
240                }
241                else if (upperMethod.Equals("PATCH", StringComparison.CurrentCultureIgnoreCase))
242                {
243                    var content = new ObjectContent<object>(body, formatter, DefaultContentType);
244                    using (var request = new HttpRequestMessage(new HttpMethod(upperMethod), fullPath) { Content = content })
245                        response = httpClient.SendAsync(request).Result;
246                }
247                else
248                {
249                    throw new ApiException(400, "Unsupported HttpMethod: " + upperMethod);
250                }
251                return new Response(response, this);
252            }
253        }

274        private async Task<string> getFullPath(string path, Dictionary<string, string> queryParams)
275        {
276            StringBuilder b = new StringBuilder();
277            b.Append("?");
278            if (queryParams != null && queryParams.Count > 0)
279            {
280                foreach(var queryParam in queryParams)
281                {
282                    b.Append(queryParam.Key);
283                    b.Append("=");
284                    b.Append(queryParam.Value);
285                    b.Append("&");
286                }
287            }
288 
289            var queryString = b.ToString();
290            await setCodebase();
291            var fullPath = _scheme + _site + "/" + _codebase + "apis/3.0" +  path + queryString.Substring(0, queryString.Length - 1);
292            return fullPath;
293        }

295        private async Task<HttpClient> getHttpClient()
296        {
297            var handler = new HttpClientHandler();
298            if (!string.IsNullOrEmpty(_connectwiseAppId))
299            {
300                if (handler.CookieContainer == null) handler.CookieContainer = new CookieContainer();
301                handler.CookieContainer.Add(new Cookie("cw-app-id", _connectwiseAppId ?? "none", "/", _site));
302            }
303            if (_usingCookieAuth)
304            {
305                await setMemberHash();
306                if (handler.CookieContainer == null) handler.CookieContainer = new CookieContainer();
307                handler.CookieContainer.Add(new Cookie("CompanyName", _companyName, "/", _site));
308                handler.CookieContainer.Add(new Cookie("MemberID", _username, "/", _site));
309                handler.CookieContainer.Add(new Cookie("MemberHash", _memberHash, "/", _site));
310            }
311            HttpClient httpClient = new HttpClient(handler, true);
312            if (!_usingCookieAuth)
313            {
314                httpClient.DefaultRequestHeaders.Authorization = new AuthenticationHeaderValue("Basic",
315                    Convert.ToBase64String(Encoding.UTF8.GetBytes(string.Format("{0}+{1}:{2}", _companyName, _publicKey, _privateKey))));
316            }
317            httpClient.DefaultRequestHeaders.Accept.Clear();
318            httpClient.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue(DefaultContentType));
319            if (_defaultHeaders != null && _defaultHeaders.Count > 0)
320                foreach(var header in _defaultHeaders)
321                    httpClient.DefaultRequestHeaders.Add(header.Key, header.Value);
322            httpClient.Timeout = _timeout ?? httpClient.Timeout;
323            return httpClient;
324        }

326        private async Task setMemberHash()
327        {
328            if (string.IsNullOrEmpty(_memberHash))
329            {
330                await setCodebase();
331                using(var httpClient = new HttpClient())
332                {
333                    var content = new MultipartFormDataContent();
334                    content.Add(new StringContent(_companyName), "CompanyName");
335                    content.Add(new StringContent(_username), "UserName");
336                    content.Add(new StringContent(_password), "Password");
337                    content.Add(new StringContent("json"), "Response");
338                    var response = await httpClient.PostAsync(_scheme + _site + "/" + _codebase + "login/Login.aspx", content);
339                    var jsonResponse = await response.Content.ReadAsStringAsync();
340                    if (string.IsNullOrEmpty(jsonResponse) || jsonResponse.Equals("CompanyFAIL", StringComparison.CurrentCultureIgnoreCase))
341                        throw new ApiException(404, "Company not found: " + _companyName);                   
342                    var jobject = JObject.Parse(jsonResponse);
343                    var successfull = Convert.ToBoolean(jobject["Success"]);
344                    if (!successfull)
345                        throw new ApiException(401, "Invalid username and password");
346                    _memberHash = Convert.ToString(jobject["Hash"]);
347                }
348            }               
349        }

351        private async Task setCodebase()
352        {
353            if (string.IsNullOrEmpty(_codebase))
354            {
355                using(var httpClient = new HttpClient())
356                {
357                    httpClient.DefaultRequestHeaders.Accept.Clear();
358                    httpClient.DefaultRequestHeaders.Accept.Add(new MediaTypeWithQualityHeaderValue("application/json"));
359                    var response = httpClient.GetAsync(_scheme + _site + "/login/companyinfo/" + _companyName).Result;
360 
361                    // Will throw an exception if not successful
362                    response.EnsureSuccessStatusCode();
363                     
364                    var jsonResponse = await response.Content.ReadAsStringAsync();
365                    if (string.IsNullOrEmpty(jsonResponse) || jsonResponse.Equals("null", StringComparison.CurrentCultureIgnoreCase))
366                        throw new ApiException(404, "Company not found: " + _companyName);
367                    var jobject = Task.Run(() => JObject.Parse(jsonResponse));
368                    _codebase = Convert.ToString(jobject.Result["Codebase"]);
369                }
370            }
371        }
